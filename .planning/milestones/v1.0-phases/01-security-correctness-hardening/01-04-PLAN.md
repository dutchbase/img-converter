    Implementation (from RESEARCH.md pattern, use 64 KB scan window not 4 KB to avoid false negatives on large animated GIFs):
    ```typescript
    export function isAnimatedGif(bytes: Uint8Array): boolean {
      // Must start with GIF magic bytes
      if (bytes[0] !== 0x47 || bytes[1] !== 0x49 || bytes[2] !== 0x46 || bytes[3] !== 0x38) {
        return false;
      }
      // Count Graphic Control Extension markers (0x21 0xF9)
      let count = 0;
      const limit = Math.min(bytes.length, 65536); // scan up to 64 KB
      for (let i = 0; i < limit - 1; i++) {
        if (bytes[i] === 0x21 && bytes[i + 1] === 0xF9) {
          count++;
          if (count > 1) return true;
        }
      }
      return false;
    }
    ```
